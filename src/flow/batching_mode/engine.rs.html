80    pub async fn handle_inserts_inner(
81        &self,
82        request: api::v1::region::InsertRequests,
83    ) -> Result<(), Error> {
84        let table_info_mgr = self.table_meta.table_info_manager();
85        let mut group_by_table_id: HashMap<TableId, Vec<api::v1::Rows>> = HashMap::new();
86
87        for r in request.requests {
88            let tid = RegionId::from(r.region_id).table_id();
89            let entry = group_by_table_id.entry(tid).or_default();
90            if let Some(rows) = r.rows {
91                entry.push(rows);
92            }
93        }
94
95        let tids = group_by_table_id.keys().cloned().collect::<Vec<TableId>>();
96        let table_infos =
97            table_info_mgr
98                .batch_get(&tids)
99                .await
100                .with_context(|_| TableNotFoundMetaSnafu {
101                    msg: format!("Failed to get table info for table ids: {:?}", tids),
102                })?;
103
104        let missing_tids = tids
105            .iter()
106            .filter(|id| !table_infos.contains_key(id))
107            .collect::<Vec<_>>();
108        if !missing_tids.is_empty() {
109            warn!(
110                "Failed to get all the table info for table ids, expected table ids: {:?}, those table doesn't exist: {:?}",
111                tids,
112                missing_tids
113            );
114        }
115
116        let group_by_table_name = group_by_table_id
117            .into_iter()
118            .filter_map(|(id, rows)| {
119                let table_name = table_infos.get(&id).map(|info| info.table_name());
120                let Some(table_name) = table_name else {
121                    warn!("Failed to get table infos for table id: {:?}", id);
122                    return None;
123                };
124                let table_name = [
125                    table_name.catalog_name,
126                    table_name.schema_name,
127                    table_name.table_name,
128                ];
129                Some((table_name, rows))
130            })
131            .collect::<HashMap<_, _>>();
132
133        let group_by_table_name = Arc::new(group_by_table_name);
134
135        let mut handles = Vec::new();
136        let tasks = self.tasks.read().await;
137        for (_flow_id, task) in tasks.iter() {
138            let src_table_names = &task.config.source_table_names;
139
140            if src_table_names
141                .iter()
142                .all(|name| !group_by_table_name.contains_key(name))
143            {
144                continue;
145            }
146
147            let group_by_table_name = group_by_table_name.clone();
148            let task = task.clone();
149
150            let handle: JoinHandle<Result<(), Error>> = tokio::spawn(async move {
151                let src_table_names = &task.config.source_table_names;
152
153                for src_table_name in src_table_names {
154                    if let Some(entry) = group_by_table_name.get(src_table_name) {
155                        let Some(expr) = &task.config.time_window_expr else {
156                            continue;
157                        };
158                        let involved_time_windows = expr.handle_rows(entry.clone()).await?;
159                        let mut state = task.state.write().unwrap();
160                        state
161                            .dirty_time_windows
162                            .add_lower_bounds(involved_time_windows.into_iter());
163                    }
164                }
165                Ok(())
166            });
167            handles.push(handle);
168        }
169
170        for handle in handles {
171            match handle.await {
172                Err(e) => {
173                    warn!("Failed to handle inserts: {e}");
174                }
175                Ok(Ok(())) => (),
176                Ok(Err(e)) => {
177                    warn!("Failed to handle inserts: {e}");
178                }
179            }
180        }
181        drop(tasks);
182
183        Ok(())
184    }
185}

187async fn get_table_name(
188    table_info: &TableInfoManager,
189    table_id: &TableId,
190) -> Result<TableName, Error> {
191    get_table_info(table_info, table_id).await.map(|info| {
192        let name = info.table_name();
193        [name.catalog_name, name.schema_name, name.table_name]
194    })
195}
196
197async fn get_table_info(
198    table_info: &TableInfoManager,
199    table_id: &TableId,
200) -> Result<TableInfoValue, Error> {
201    table_info
202        .get(*table_id)
203        .await
204        .map_err(BoxedError::new)
205        .context(ExternalSnafu)?
206        .with_context(|| UnexpectedSnafu {
207            reason: format!("Table id = {:?}, couldn't found table name", table_id),
208        })
209        .map(|info| info.into_inner())
210}

212impl BatchingEngine {
213    pub async fn create_flow_inner(&self, args: CreateFlowArgs) -> Result<Option<FlowId>, Error> {
214        let CreateFlowArgs {
215            flow_id,
216            sink_table_name,
217            source_table_ids,
218            create_if_not_exists,
219            or_replace,
220            expire_after,
221            comment: _,
222            sql,
223            flow_options,
224            query_ctx,
225        } = args;
226
227        // or replace logic
228        {
229            let is_exist = self.tasks.read().await.contains_key(&flow_id);
230            match (create_if_not_exists, or_replace, is_exist) {
231                // if replace, ignore that old flow exists
232                (_, true, true) => {
233                    info!("Replacing flow with id={}", flow_id);
234                }
235                (false, false, true) => FlowAlreadyExistSnafu { id: flow_id }.fail()?,
236                // already exists, and not replace, return None
237                (true, false, true) => {
238                    info!("Flow with id={} already exists, do nothing", flow_id);
239                    return Ok(None);
240                }
241
242                // continue as normal
243                (_, _, false) => (),
244            }
245        }
246
247        let flow_type = flow_options.get(FlowType::FLOW_TYPE_KEY);
248
249        ensure!(
250            match flow_type {
251                None => true,
252                Some(ty) if ty == FlowType::BATCHING => true,
253                _ => false,
254            },
255            UnexpectedSnafu {
256                reason: format!("Flow type is not batching nor None, got {flow_type:?}")
257            }
258        );
259
260        let Some(query_ctx) = query_ctx else {
261            UnexpectedSnafu {
262                reason: "Query context is None".to_string(),
263            }
264            .fail()?
265        };
266        let query_ctx = Arc::new(query_ctx);
267        let mut source_table_names = Vec::with_capacity(2);
268        for src_id in source_table_ids {
269            // also check table option to see if ttl!=instant
270            let table_name = get_table_name(self.table_meta.table_info_manager(), &src_id).await?;
271            let table_info = get_table_info(self.table_meta.table_info_manager(), &src_id).await?;
272            ensure!(
273                table_info.table_info.meta.options.ttl != Some(TimeToLive::Instant),
274                UnsupportedSnafu {
275                    reason: format!(
276                        "Source table `{}`(id={}) has instant TTL, Instant TTL is not supported under batching mode. Consider using a TTL longer than flush interval",
277                        table_name.join("."),
278                        src_id
279                    ),
280                }
281            );
282
283            source_table_names.push(table_name);
284        }
285
286        let (tx, rx) = oneshot::channel();
287
288        let plan = sql_to_df_plan(query_ctx.clone(), self.query_engine.clone(), &sql, true).await?;
289        let (column_name, time_window_expr, _, df_schema) = find_time_window_expr(
290            &plan,
291            self.query_engine.engine_state().catalog_manager().clone(),
292            query_ctx.clone(),
293        )
294        .await?;
295
296        let phy_expr = time_window_expr
297            .map(|expr| {
298                TimeWindowExpr::from_expr(
299                    &expr,
300                    &column_name,
301                    &df_schema,
302                    &self.query_engine.engine_state().session_state(),
303                )
304            })
305            .transpose()?;
306
307        debug!(
308            "Flow id={}, found time window expr={}",
309            flow_id,
310            phy_expr
311                .as_ref()
312                .map(|phy_expr| phy_expr.to_string())
313                .unwrap_or("None".to_string())
314        );
315
316        let task = BatchingTask::try_new(
317            flow_id,
318            &sql,
319            plan,
320            phy_expr,
321            expire_after,
322            sink_table_name,
323            source_table_names,
324            query_ctx,
325            self.catalog_manager.clone(),
326            rx,
327        )?;
328
329        let task_inner = task.clone();
330        let engine = self.query_engine.clone();
331        let frontend = self.frontend_client.clone();
332
333        // check execute once first to detect any error early
334        task.check_or_create_sink_table(&engine, &frontend).await?;
335
336        // TODO(discord9): use time wheel or what for better
337        let handle = common_runtime::spawn_global(async move {
338            task_inner.start_executing_loop(engine, frontend).await;
339        });
340        task.state.write().unwrap().task_handle = Some(handle);
341
342        // only replace here not earlier because we want the old one intact if something went wrong before this line
343        let replaced_old_task_opt = self.tasks.write().await.insert(flow_id, task);
344        drop(replaced_old_task_opt);
345
346        self.shutdown_txs.write().await.insert(flow_id, tx);
347
348        Ok(Some(flow_id))
349    }

351    pub async fn remove_flow_inner(&self, flow_id: FlowId) -> Result<(), Error> {
352        if self.tasks.write().await.remove(&flow_id).is_none() {
353            warn!("Flow {flow_id} not found in tasks");
354            FlowNotFoundSnafu { id: flow_id }.fail()?;
355        }
356        let Some(tx) = self.shutdown_txs.write().await.remove(&flow_id) else {
357            UnexpectedSnafu {
358                reason: format!("Can't found shutdown tx for flow {flow_id}"),
359            }
360            .fail()?
361        };
362        if tx.send(()).is_err() {
363            warn!("Fail to shutdown flow {flow_id} due to receiver already dropped, maybe flow {flow_id} is already dropped?")
364        }
365        Ok(())
366    }
367
368    pub async fn flush_flow_inner(&self, flow_id: FlowId) -> Result<usize, Error> {
369        debug!("Try flush flow {flow_id}");
370        let task = self.tasks.read().await.get(&flow_id).cloned();
371        let task = task.with_context(|| FlowNotFoundSnafu { id: flow_id })?;
372
373        task.mark_all_windows_as_dirty()?;
374
375        let res = task
376            .gen_exec_once(&self.query_engine, &self.frontend_client)
377            .await?;
378
379        let affected_rows = res.map(|(r, _)| r).unwrap_or_default() as usize;
380        debug!(
381            "Successfully flush flow {flow_id}, affected rows={}",
382            affected_rows
383        );
384        Ok(affected_rows)
385    }
386
387    /// Determine if the batching mode flow task exists with given flow id
388    pub async fn flow_exist_inner(&self, flow_id: FlowId) -> bool {
389        self.tasks.read().await.contains_key(&flow_id)
390    }
391}

393impl FlowEngine for BatchingEngine {
394    async fn create_flow(&self, args: CreateFlowArgs) -> Result<Option<FlowId>, Error> {
395        self.create_flow_inner(args).await
396    }
397    async fn remove_flow(&self, flow_id: FlowId) -> Result<(), Error> {
398        self.remove_flow_inner(flow_id).await
399    }
400    async fn flush_flow(&self, flow_id: FlowId) -> Result<usize, Error> {
401        self.flush_flow_inner(flow_id).await
402    }
403    async fn flow_exist(&self, flow_id: FlowId) -> Result<bool, Error> {
404        Ok(self.flow_exist_inner(flow_id).await)
405    }
406    async fn list_flows(&self) -> Result<impl IntoIterator<Item = FlowId>, Error> {
407        Ok(self.tasks.read().await.keys().cloned().collect::<Vec<_>>())
408    }
409    async fn handle_flow_inserts(
410        &self,
411        request: api::v1::region::InsertRequests,
412    ) -> Result<(), Error> {
413        self.handle_inserts_inner(request).await
414    }
