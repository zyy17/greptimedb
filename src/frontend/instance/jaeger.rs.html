54#[async_trait]
55impl JaegerQueryHandler for Instance {
56    async fn get_services(&self, ctx: QueryContextRef) -> ServerResult<Output> {
57        // It's equivalent to `SELECT DISTINCT(service_name) FROM {db}.{trace_table}`.
58        Ok(query_trace_table(
59            ctx,
60            self.catalog_manager(),
61            self.query_engine(),
62            vec![col(SERVICE_NAME_COLUMN)],
63            vec![],
64            vec![],
65            None,
66            None,
67            vec![col(SERVICE_NAME_COLUMN)],
68        )
69        .await?)
70    }
71
72    async fn get_operations(
73        &self,
74        ctx: QueryContextRef,
75        service_name: &str,
76        span_kind: Option<&str>,
77        start_time: Option<i64>,
78        end_time: Option<i64>,
79    ) -> ServerResult<Output> {
80        let mut filters = vec![col(SERVICE_NAME_COLUMN).eq(lit(service_name))];
81
82        if let Some(span_kind) = span_kind {
83            filters.push(col(SPAN_KIND_COLUMN).eq(lit(format!(
84                "{}{}",
85                SPAN_KIND_PREFIX,
86                span_kind.to_uppercase()
87            ))));
88        }
89
90        if let Some(start_time) = start_time {
91            // Microseconds to nanoseconds.
92            filters.push(col(TIMESTAMP_COLUMN).gt_eq(lit_timestamp_nano(start_time * 1_000)));
93        }
94
95        if let Some(end_time) = end_time {
96            // Microseconds to nanoseconds.
97            filters.push(col(TIMESTAMP_COLUMN).lt_eq(lit_timestamp_nano(end_time * 1_000)));
98        }
99
100        // It's equivalent to
101        //
102        // ```
103        // SELECT DISTINCT span_name, span_kind
104        // FROM
105        //   {db}.{trace_table}
106        // WHERE
107        //   service_name = '{service_name}' AND
108        //   timestamp >= {start_time} AND
109        //   timestamp <= {end_time} AND
110        //   span_kind = '{span_kind}'
111        // ORDER BY
112        //   span_name ASC
113        // ```.
114        Ok(query_trace_table(
115            ctx,
116            self.catalog_manager(),
117            self.query_engine(),
118            vec![
119                col(SPAN_NAME_COLUMN),
120                col(SPAN_KIND_COLUMN),
121                col(SERVICE_NAME_COLUMN),
122                col(TIMESTAMP_COLUMN),
123            ],
124            filters,
125            vec![col(SPAN_NAME_COLUMN).sort(true, false)], // Sort by span_name in ascending order.
126            Some(DEFAULT_LIMIT),
127            None,
128            vec![col(SPAN_NAME_COLUMN), col(SPAN_KIND_COLUMN)],
129        )
130        .await?)
131    }
132
133    async fn get_trace(
134        &self,
135        ctx: QueryContextRef,
136        trace_id: &str,
137        start_time: Option<i64>,
138        end_time: Option<i64>,
139    ) -> ServerResult<Output> {
140        // It's equivalent to
141        //
142        // ```
143        // SELECT
144        //   *
145        // FROM
146        //   {db}.{trace_table}
147        // WHERE
148        //   trace_id = '{trace_id}' AND
149        //   timestamp >= {start_time} AND
150        //   timestamp <= {end_time}
151        // ORDER BY
152        //   timestamp DESC
153        // ```.
154        let selects = vec![wildcard()];
155
156        let mut filters = vec![col(TRACE_ID_COLUMN).eq(lit(trace_id))];
157
158        if let Some(start_time) = start_time {
159            // Microseconds to nanoseconds.
160            filters.push(col(TIMESTAMP_COLUMN).gt_eq(lit_timestamp_nano(start_time * 1_000)));
161        }
162
163        if let Some(end_time) = end_time {
164            // Microseconds to nanoseconds.
165            filters.push(col(TIMESTAMP_COLUMN).lt_eq(lit_timestamp_nano(end_time * 1_000)));
166        }
167
168        Ok(query_trace_table(
169            ctx,
170            self.catalog_manager(),
171            self.query_engine(),
172            selects,
173            filters,
174            vec![col(TIMESTAMP_COLUMN).sort(false, false)], // Sort by timestamp in descending order.
175            Some(DEFAULT_LIMIT),
176            None,
177            vec![],
178        )
179        .await?)
180    }
181
182    async fn find_traces(
183        &self,
184        ctx: QueryContextRef,
185        query_params: QueryTraceParams,
186    ) -> ServerResult<Output> {
187        let selects = vec![wildcard()];
188
189        let mut filters = vec![];
190
191        if let Some(operation_name) = query_params.operation_name {
192            filters.push(col(SPAN_NAME_COLUMN).eq(lit(operation_name)));
193        }
194
195        if let Some(start_time) = query_params.start_time {
196            filters.push(col(TIMESTAMP_COLUMN).gt_eq(lit_timestamp_nano(start_time)));
197        }
198
199        if let Some(end_time) = query_params.end_time {
200            filters.push(col(TIMESTAMP_COLUMN).lt_eq(lit_timestamp_nano(end_time)));
201        }
202
203        if let Some(min_duration) = query_params.min_duration {
204            filters.push(col(DURATION_NANO_COLUMN).gt_eq(lit(min_duration)));
205        }
206
207        if let Some(max_duration) = query_params.max_duration {
208            filters.push(col(DURATION_NANO_COLUMN).lt_eq(lit(max_duration)));
209        }
210
211        Ok(query_trace_table(
212            ctx,
213            self.catalog_manager(),
214            self.query_engine(),
215            selects,
216            filters,
217            vec![col(TIMESTAMP_COLUMN).sort(false, false)], // Sort by timestamp in descending order.
218            Some(DEFAULT_LIMIT),
219            query_params.tags,
220            vec![],
221        )
222        .await?)
223    }
224}

226#[allow(clippy::too_many_arguments)]
227async fn query_trace_table(
228    ctx: QueryContextRef,
229    catalog_manager: &CatalogManagerRef,
230    query_engine: &QueryEngineRef,
231    selects: Vec<Expr>,
232    filters: Vec<Expr>,
233    sorts: Vec<SortExpr>,
234    limit: Option<usize>,
235    tags: Option<HashMap<String, JsonValue>>,
236    distincts: Vec<Expr>,
237) -> ServerResult<Output> {
238    let trace_table_name = ctx
239        .extension(JAEGER_QUERY_TABLE_NAME_KEY)
240        .unwrap_or(TRACE_TABLE_NAME);
241
242    // If only select services, use the trace services table.
243    let table_name = {
244        if selects.len() == 1 && selects[0] == col(SERVICE_NAME_COLUMN) {
245            &trace_services_table_name(trace_table_name)
246        } else {
247            trace_table_name
248        }
249    };
250
251    let table = catalog_manager
252        .table(
253            ctx.current_catalog(),
254            &ctx.current_schema(),
255            table_name,
256            Some(&ctx),
257        )
258        .await
259        .context(CatalogSnafu)?
260        .with_context(|| TableNotFoundSnafu {
261            table: table_name,
262            catalog: ctx.current_catalog(),
263            schema: ctx.current_schema(),
264        })?;
265
266    let is_data_model_v1 = table
267        .table_info()
268        .meta
269        .options
270        .extra_options
271        .get(TABLE_DATA_MODEL)
272        .map(|s| s.as_str())
273        == Some(TABLE_DATA_MODEL_TRACE_V1);
274
275    let df_context = create_df_context(query_engine, ctx.clone())?;
276
277    let dataframe = df_context
278        .read_table(Arc::new(DfTableProviderAdapter::new(table)))
279        .context(DataFusionSnafu)?;
280
281    let dataframe = dataframe.select(selects).context(DataFusionSnafu)?;
282
283    // Apply all filters.
284    let dataframe = filters
285        .into_iter()
286        .chain(tags.map_or(Ok(vec![]), |t| {
287            tags_filters(&dataframe, t, is_data_model_v1)
288        })?)
289        .try_fold(dataframe, |df, expr| {
290            df.filter(expr).context(DataFusionSnafu)
291        })?;
292
293    // Apply the distinct if needed.
294    let dataframe = if !distincts.is_empty() {
295        dataframe
296            .distinct_on(distincts.clone(), distincts, None)
297            .context(DataFusionSnafu)?
298    } else {
299        dataframe
300    };
301
302    // Apply the sorts if needed.
303    let dataframe = if !sorts.is_empty() {
304        dataframe.sort(sorts).context(DataFusionSnafu)?
305    } else {
306        dataframe
307    };
308
309    // Apply the limit if needed.
310    let dataframe = if let Some(limit) = limit {
311        dataframe.limit(0, Some(limit)).context(DataFusionSnafu)?
312    } else {
313        dataframe
314    };
315
316    // Execute the query and collect the result.
317    let stream = dataframe.execute_stream().await.context(DataFusionSnafu)?;
318
319    let output = Output::new_with_stream(Box::pin(
320        RecordBatchStreamAdapter::try_new(stream).context(CollectRecordbatchSnafu)?,
321    ));
322
323    Ok(output)
324}

326// The current implementation registers UDFs during the planning stage, which makes it difficult
327// to utilize them through DataFrame APIs. To address this limitation, we create a new session
328// context and register the required UDFs, allowing them to be decoupled from the global context.
329// TODO(zyy17): Is it possible or necessary to reuse the existing session context?
330fn create_df_context(
331    query_engine: &QueryEngineRef,
332    ctx: QueryContextRef,
333) -> ServerResult<SessionContext> {
334    let df_context = SessionContext::new_with_state(
335        SessionStateBuilder::new_from_existing(query_engine.engine_state().session_state()).build(),
336    );
337
338    // The following JSON UDFs will be used for tags filters on v0 data model.
339    let udfs: Vec<FunctionRef> = vec![
340        Arc::new(JsonGetInt),
341        Arc::new(JsonGetFloat),
342        Arc::new(JsonGetBool),
343        Arc::new(JsonGetString),
344    ];
345
346    for udf in udfs {
347        df_context.register_udf(create_udf(
348            udf,
349            ctx.clone(),
350            Arc::new(FunctionState::default()),
351        ));
352    }
353
354    Ok(df_context)
355}

357fn json_tag_filters(
358    dataframe: &DataFrame,
359    tags: HashMap<String, JsonValue>,
360) -> ServerResult<Vec<Expr>> {
361    let mut filters = vec![];
362
363    // NOTE: The key of the tags may contain `.`, for example: `http.status_code`, so we need to use `["http.status_code"]` in json path to access the value.
364    for (key, value) in tags.iter() {
365        if let JsonValue::String(value) = value {
366            filters.push(
367                dataframe
368                    .registry()
369                    .udf(JsonGetString {}.name())
370                    .context(DataFusionSnafu)?
371                    .call(vec![
372                        col(SPAN_ATTRIBUTES_COLUMN),
373                        lit(format!("[\"{}\"]", key)),
374                    ])
375                    .eq(lit(value)),
376            );
377        }
378        if let JsonValue::Number(value) = value {
379            if value.is_i64() {
380                filters.push(
381                    dataframe
382                        .registry()
383                        .udf(JsonGetInt {}.name())
384                        .context(DataFusionSnafu)?
385                        .call(vec![
386                            col(SPAN_ATTRIBUTES_COLUMN),
387                            lit(format!("[\"{}\"]", key)),
388                        ])
389                        .eq(lit(value.as_i64().unwrap())),
390                );
391            }
392            if value.is_f64() {
393                filters.push(
394                    dataframe
395                        .registry()
396                        .udf(JsonGetFloat {}.name())
397                        .context(DataFusionSnafu)?
398                        .call(vec![
399                            col(SPAN_ATTRIBUTES_COLUMN),
400                            lit(format!("[\"{}\"]", key)),
401                        ])
402                        .eq(lit(value.as_f64().unwrap())),
403                );
404            }
405        }
406        if let JsonValue::Bool(value) = value {
407            filters.push(
408                dataframe
409                    .registry()
410                    .udf(JsonGetBool {}.name())
411                    .context(DataFusionSnafu)?
412                    .call(vec![
413                        col(SPAN_ATTRIBUTES_COLUMN),
414                        lit(format!("[\"{}\"]", key)),
415                    ])
416                    .eq(lit(*value)),
417            );
418        }
419    }
420
421    Ok(filters)
422}

424fn flatten_tag_filters(tags: HashMap<String, JsonValue>) -> ServerResult<Vec<Expr>> {
425    let filters = tags
426        .into_iter()
427        .filter_map(|(key, value)| {
428            let key = format!("\"span_attributes.{}\"", key);
429            match value {
430                JsonValue::String(value) => Some(col(key).eq(lit(value))),
431                JsonValue::Number(value) => {
432                    if value.is_f64() {
433                        // safe to unwrap as checked previously
434                        Some(col(key).eq(lit(value.as_f64().unwrap())))
435                    } else {
436                        Some(col(key).eq(lit(value.as_i64().unwrap())))
437                    }
438                }
439                JsonValue::Bool(value) => Some(col(key).eq(lit(value))),
440                JsonValue::Null => Some(col(key).is_null()),
441                // not supported at the moment
442                JsonValue::Array(_value) => None,
443                JsonValue::Object(_value) => None,
444            }
445        })
446        .collect();
447    Ok(filters)
448}
449
450fn tags_filters(
451    dataframe: &DataFrame,
452    tags: HashMap<String, JsonValue>,
453    is_data_model_v1: bool,
454) -> ServerResult<Vec<Expr>> {
455    if is_data_model_v1 {
456        flatten_tag_filters(tags)
457    } else {
458        json_tag_filters(dataframe, tags)
459    }
