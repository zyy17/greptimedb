102    pub async fn new(cluster_name: &str) -> Self {
103        let endpoints = env::var("GT_ETCD_ENDPOINTS").unwrap_or_default();
104
105        let kv_backend: KvBackendRef = if endpoints.is_empty() {
106            Arc::new(MemoryKvBackend::new())
107        } else {
108            let endpoints = endpoints
109                .split(',')
110                .map(|s| s.to_string())
111                .collect::<Vec<String>>();
112            let backend = EtcdStore::with_endpoints(endpoints, 128)
113                .await
114                .expect("malformed endpoints");
115            // Each retry requires a new isolation namespace.
116            let chroot = format!("{}{}", cluster_name, Uuid::new_v4());
117            Arc::new(ChrootKvBackend::new(chroot.into(), backend))
118        };
119
120        Self {
121            cluster_name: cluster_name.to_string(),
122            kv_backend,
123            store_config: None,
124            store_providers: None,
125            datanodes: None,
126            datanode_wal_config: DatanodeWalConfig::default(),
127            metasrv_wal_config: MetasrvWalConfig::default(),
128            shared_home_dir: None,
129            meta_selector: None,
130        }
131    }

175    pub async fn build_with(
176        &self,
177        datanode_options: Vec<DatanodeOptions>,
178        guards: Vec<TestGuard>,
179    ) -> GreptimeDbCluster {
180        let datanodes = datanode_options.len();
181        let channel_config = ChannelConfig::new().timeout(Duration::from_secs(20));
182        let datanode_clients = Arc::new(NodeClients::new(channel_config));
183
184        let opt = MetasrvOptions {
185            procedure: ProcedureConfig {
186                // Due to large network delay during cross data-center.
187                // We only make max_retry_times and retry_delay large than the default in tests.
188                max_retry_times: 5,
189                retry_delay: Duration::from_secs(1),
190                max_metadata_value_size: None,
191                max_running_procedures: 128,
192            },
193            wal: self.metasrv_wal_config.clone(),
194            grpc: GrpcOptions {
195                server_addr: "127.0.0.1:3002".to_string(),
196                ..Default::default()
197            },
198            ..Default::default()
199        };
200
201        let metasrv = meta_srv::mocks::mock(
202            opt,
203            self.kv_backend.clone(),
204            self.meta_selector.clone(),
205            Some(datanode_clients.clone()),
206            None,
207        )
208        .await;
209
210        let datanode_instances = self
211            .build_datanodes_with_options(&metasrv, &datanode_options)
212            .await;
213
214        build_datanode_clients(datanode_clients.clone(), &datanode_instances, datanodes).await;
215
216        self.wait_datanodes_alive(metasrv.metasrv.meta_peer_client(), datanodes)
217            .await;
218
219        let mut frontend = self.build_frontend(metasrv.clone(), datanode_clients).await;
220
221        test_util::prepare_another_catalog_and_schema(&frontend.instance).await;
222
223        frontend.start().await.unwrap();
224
225        GreptimeDbCluster {
226            datanode_options,
227            guards,
228            datanode_instances,
229            kv_backend: self.kv_backend.clone(),
230            metasrv: metasrv.metasrv,
231            frontend: Arc::new(frontend),
232        }
233    }

235    pub async fn build(&self) -> GreptimeDbCluster {
236        let datanodes = self.datanodes.unwrap_or(4);
237        let (datanode_options, guards) = self.build_datanode_options_and_guards(datanodes).await;
238        self.build_with(datanode_options, guards).await
239    }

241    async fn build_datanode_options_and_guards(
242        &self,
243        datanodes: u32,
244    ) -> (Vec<DatanodeOptions>, Vec<TestGuard>) {
245        let mut options = Vec::with_capacity(datanodes as usize);
246        let mut guards = Vec::with_capacity(datanodes as usize);
247
248        for i in 0..datanodes {
249            let datanode_id = i as u64 + 1;
250            let mut opts = if let Some(store_config) = &self.store_config {
251                let home_dir = if let Some(home_dir) = &self.shared_home_dir {
252                    home_dir.path().to_str().unwrap().to_string()
253                } else {
254                    let home_tmp_dir = create_temp_dir(&format!("gt_home_{}", &self.cluster_name));
255                    let home_dir = home_tmp_dir.path().to_str().unwrap().to_string();
256                    guards.push(TestGuard {
257                        home_guard: FileDirGuard::new(home_tmp_dir),
258                        storage_guards: Vec::new(),
259                    });
260
261                    home_dir
262                };
263
264                create_datanode_opts(
265                    store_config.clone(),
266                    vec![],
267                    home_dir,
268                    self.datanode_wal_config.clone(),
269                )
270            } else {
271                let (opts, guard) = create_tmp_dir_and_datanode_opts(
272                    StorageType::File,
273                    self.store_providers.clone().unwrap_or_default(),
274                    &format!("{}-dn-{}", self.cluster_name, datanode_id),
275                    self.datanode_wal_config.clone(),
276                );
277                guards.push(guard);
278
279                opts
280            };
281            opts.node_id = Some(datanode_id);
282
283            options.push(opts);
284        }
285        (options, guards)
286    }

288    async fn build_datanodes_with_options(
289        &self,
290        metasrv: &MockInfo,
291        options: &[DatanodeOptions],
292    ) -> HashMap<DatanodeId, Datanode> {
293        let mut instances = HashMap::with_capacity(options.len());
294

296            let datanode = self.create_datanode(opts.clone(), metasrv.clone()).await;

303    async fn wait_datanodes_alive(
304        &self,
305        meta_peer_client: &MetaPeerClientRef,
306        expected_datanodes: usize,
307    ) {
308        for _ in 0..10 {
309            let alive_datanodes = meta_srv::lease::alive_datanodes(meta_peer_client, u64::MAX)
310                .await
311                .unwrap()
312                .len();
313            if alive_datanodes == expected_datanodes {
314                return;
315            }
316            tokio::time::sleep(Duration::from_secs(1)).await
317        }
318        panic!("Some Datanodes are not alive in 10 seconds!")
319    }

321    async fn create_datanode(&self, opts: DatanodeOptions, metasrv: MockInfo) -> Datanode {
322        let mut meta_client = MetaClientBuilder::datanode_default_options(opts.node_id.unwrap())
323            .channel_manager(metasrv.channel_manager)
324            .build();
325        meta_client.start(&[&metasrv.server_addr]).await.unwrap();
326        let meta_client = Arc::new(meta_client);
327
328        let meta_backend = Arc::new(MetaKvBackend {
329            client: meta_client.clone(),
330        });
331
332        let layered_cache_registry = Arc::new(
333            LayeredCacheRegistryBuilder::default()
334                .add_cache_registry(build_datanode_cache_registry(meta_backend.clone()))
335                .build(),
336        );
337
338        let mut builder = DatanodeBuilder::new(opts, Plugins::default(), meta_backend);
339        builder
340            .with_cache_registry(layered_cache_registry)
341            .with_meta_client(meta_client);
342        let mut datanode = builder.build().await.unwrap();
343
344        datanode.start_heartbeat().await.unwrap();
345
346        datanode
347    }

349    async fn build_frontend(
350        &self,
351        metasrv: MockInfo,
352        datanode_clients: Arc<NodeClients>,
353    ) -> Frontend {
354        let mut meta_client = MetaClientBuilder::frontend_default_options()
355            .channel_manager(metasrv.channel_manager)
356            .enable_access_cluster_info()
357            .build();
358        meta_client.start(&[&metasrv.server_addr]).await.unwrap();
359        let meta_client = Arc::new(meta_client);
360
361        let cached_meta_backend = Arc::new(
362            CachedKvBackendBuilder::new(Arc::new(MetaKvBackend::new(meta_client.clone()))).build(),
363        );
364
365        let layered_cache_builder = LayeredCacheRegistryBuilder::default().add_cache_registry(
366            CacheRegistryBuilder::default()
367                .add_cache(cached_meta_backend.clone())
368                .build(),
369        );
370        let fundamental_cache_registry =
371            build_fundamental_cache_registry(Arc::new(MetaKvBackend::new(meta_client.clone())));
372        let cache_registry = Arc::new(
373            with_default_composite_cache_registry(
374                layered_cache_builder.add_cache_registry(fundamental_cache_registry),
375            )
376            .unwrap()
377            .build(),
378        );
379
380        let information_extension =
381            Arc::new(DistributedInformationExtension::new(meta_client.clone()));
382        let catalog_manager = KvBackendCatalogManager::new(
383            information_extension,
384            cached_meta_backend.clone(),
385            cache_registry.clone(),
386            None,
387        );
388
389        let handlers_executor = HandlerGroupExecutor::new(vec![
390            Arc::new(ParseMailboxMessageHandler),
391            Arc::new(InvalidateCacheHandler::new(cache_registry.clone())),
392        ]);
393
394        let options = FrontendOptions::default();
395        let heartbeat_task = HeartbeatTask::new(
396            &options,
397            meta_client.clone(),
398            HeartbeatOptions::default(),
399            Arc::new(handlers_executor),
400        );
401
402        let instance = FrontendBuilder::new(
403            options,
404            cached_meta_backend.clone(),
405            cache_registry.clone(),
406            catalog_manager,
407            datanode_clients,
408            meta_client,
409        )
410        .with_local_cache_invalidator(cache_registry)
411        .try_build()
412        .await
413        .unwrap();
414        let instance = Arc::new(instance);
415
416        Frontend {
417            instance,
418            servers: ServerHandlers::default(),
419            heartbeat_task: Some(heartbeat_task),
420            export_metrics_task: None,
421        }
422    }

425async fn build_datanode_clients(
426    clients: Arc<NodeClients>,
427    instances: &HashMap<DatanodeId, Datanode>,
428    datanodes: usize,
429) {
430    for i in 0..datanodes {
431        let datanode_id = i as u64 + 1;
432        let instance = instances.get(&datanode_id).unwrap();
433        let (addr, client) = create_datanode_client(instance).await;
434        clients
435            .insert_client(Peer::new(datanode_id, addr), client)
436            .await;
437    }
438}

440async fn create_datanode_client(datanode: &Datanode) -> (String, Client) {
441    let (client, server) = tokio::io::duplex(1024);
442
443    let runtime = RuntimeBuilder::default()
444        .worker_threads(2)
445        .thread_name("grpc-handlers")
446        .build()
447        .unwrap();
448
449    let flight_handler = FlightCraftWrapper(datanode.region_server());
450
451    let region_server_handler =
452        RegionServerRequestHandler::new(Arc::new(datanode.region_server()), runtime);
453
454    let _handle = tokio::spawn(async move {
455        Server::builder()
456            .add_service(
457                FlightServiceServer::new(flight_handler)
458                    .accept_compressed(CompressionEncoding::Gzip)
459                    .accept_compressed(CompressionEncoding::Zstd)
460                    .send_compressed(CompressionEncoding::Gzip)
461                    .send_compressed(CompressionEncoding::Zstd),
462            )
463            .add_service(
464                RegionServer::new(region_server_handler)
465                    .accept_compressed(CompressionEncoding::Gzip)
466                    .accept_compressed(CompressionEncoding::Zstd)
467                    .send_compressed(CompressionEncoding::Gzip)
468                    .send_compressed(CompressionEncoding::Zstd),
469            )
470            .serve_with_incoming(futures::stream::iter(vec![Ok::<_, std::io::Error>(server)]))
471            .await
472    });
473
474    // Move client to an option so we can _move_ the inner value
475    // on the first attempt to connect. All other attempts will fail.
476    let mut client = Some(client);
477    // `PEER_PLACEHOLDER_ADDR` is just a placeholder, does not actually connect to it.
478    let addr = PEER_PLACEHOLDER_ADDR;
479    let channel_manager = ChannelManager::new();
480    let _ = channel_manager
481        .reset_with_connector(
482            addr,
483            service_fn(move |_| {
484                let client = client.take();
485
486                async move {
487                    if let Some(client) = client {
488                        Ok(TokioIo::new(client))
489                    } else {
490                        Err(std::io::Error::other("Client already taken"))
491                    }
492                }
493            }),
494        )
495        .unwrap();
496    (
497        addr.to_string(),
498        Client::with_manager_and_urls(channel_manager, vec![addr]),
499    )
