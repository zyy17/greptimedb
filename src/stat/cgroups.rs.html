17use std::fs::read_to_string;
18use std::path::Path;
19
20#[cfg(target_os = "linux")]
21use nix::sys::{statfs, statfs::statfs};
22
23/// `MAX_VALUE` is used to indicate that the resource is unlimited.
24pub const MAX_VALUE: i64 = -1;
25
26const CGROUP_UNIFIED_MOUNTPOINT: &str = "/sys/fs/cgroup";
27
28const MEMORY_MAX_FILE_CGROUP_V2: &str = "memory.max";
29const MEMORY_MAX_FILE_CGROUP_V1: &str = "memory.limit_in_bytes";
30const CPU_MAX_FILE_CGROUP_V2: &str = "cpu.max";
31const CPU_QUOTA_FILE_CGROUP_V1: &str = "cpu.cfs_quota_us";
32const CPU_PERIOD_FILE_CGROUP_V1: &str = "cpu.cfs_period_us";
33
34// `MAX_VALUE_CGROUP_V2` string in `/sys/fs/cgroup/cpu.max` and `/sys/fs/cgroup/memory.max` to indicate that the resource is unlimited.
35const MAX_VALUE_CGROUP_V2: &str = "max";
36
37// For cgroup v1, if the memory is unlimited, it will return a very large value(different from platform) that close to 2^63.
38// For easier comparison, if the memory limit is larger than 1PB we consider it as unlimited.
39const MAX_MEMORY_IN_BYTES: i64 = 1125899906842624; // 1PB
40
41/// Get the limit of memory in bytes.
42///
43/// - If the memory is unlimited, return `-1`.
44/// - Return `None` if it fails to read the memory limit or not on linux.
45pub fn get_memory_limit() -> Option<i64> {
46    #[cfg(target_os = "linux")]
47    {
48        let memory_max_file = if is_cgroup_v2()? {
49            // Read `/sys/fs/cgroup/memory.max` to get the memory limit.
50            MEMORY_MAX_FILE_CGROUP_V2
51        } else {
52            // Read `/sys/fs/cgroup/memory.limit_in_bytes` to get the memory limit.
53            MEMORY_MAX_FILE_CGROUP_V1
54        };
55
56        // For cgroup v1, it will return a very large value(different from platform) if the memory is unlimited.
57        let memory_limit =
58            read_value_from_file(Path::new(CGROUP_UNIFIED_MOUNTPOINT).join(memory_max_file))?;
59
60        // If memory limit exceeds 1PB(cgroup v1), consider it as unlimited.
61        if memory_limit > MAX_MEMORY_IN_BYTES {
62            return Some(MAX_VALUE);
63        }
64        Some(memory_limit)
65    }
66
67    #[cfg(not(target_os = "linux"))]
68    None
69}
70
71/// Get the limit of cpu in millicores.
72///
73/// - If the cpu is unlimited, return `-1`.
74/// - Return `None` if it fails to read the cpu limit or not on linux.
75pub fn get_cpu_limit() -> Option<i64> {
76    #[cfg(target_os = "linux")]
77    if is_cgroup_v2()? {
78        // Read `/sys/fs/cgroup/cpu.max` to get the cpu limit.
79        get_cgroup_v2_cpu_limit(Path::new(CGROUP_UNIFIED_MOUNTPOINT).join(CPU_MAX_FILE_CGROUP_V2))
80    } else {
81        // Read `/sys/fs/cgroup/cpu.cfs_quota_us` and `/sys/fs/cgroup/cpu.cfs_period_us` to get the cpu limit.
82        let quota = read_value_from_file(
83            Path::new(CGROUP_UNIFIED_MOUNTPOINT).join(CPU_QUOTA_FILE_CGROUP_V1),
84        )?;
85
86        if quota == MAX_VALUE {
87            return Some(MAX_VALUE);
88        }
89
90        let period = read_value_from_file(
91            Path::new(CGROUP_UNIFIED_MOUNTPOINT).join(CPU_PERIOD_FILE_CGROUP_V1),
92        )?;
93
94        // Return the cpu limit in millicores.
95        Some(quota * 1000 / period)
96    }
97
98    #[cfg(not(target_os = "linux"))]
99    None
100}

102// Check whether the cgroup is v2.
103// - Return `true` if the cgroup is v2, otherwise return `false`.
104// - Return `None` if the detection fails or not on linux.
105fn is_cgroup_v2() -> Option<bool> {
106    #[cfg(target_os = "linux")]
107    {
108        let path = Path::new(CGROUP_UNIFIED_MOUNTPOINT);
109        let fs_stat = statfs(path).ok()?;
110        Some(fs_stat.filesystem_type() == statfs::CGROUP2_SUPER_MAGIC)
111    }
112
113    #[cfg(not(target_os = "linux"))]
114    None
115}
116
117fn read_value_from_file<P: AsRef<Path>>(path: P) -> Option<i64> {
118    let content = read_to_string(&path).ok()?;
119
120    // If the content starts with "max", return `MAX_VALUE`.
121    if content.starts_with(MAX_VALUE_CGROUP_V2) {
122        return Some(MAX_VALUE);
123    }
124
125    content.trim().parse::<i64>().ok()
126}
127
128fn get_cgroup_v2_cpu_limit<P: AsRef<Path>>(path: P) -> Option<i64> {
129    let content = read_to_string(&path).ok()?;
130
131    let fields = content.trim().split(' ').collect::<Vec<&str>>();
132    if fields.len() != 2 {
133        return None;
134    }
135
136    // If the cpu is unlimited, it will be `-1`.
137    let quota = fields[0].trim();
138    if quota == MAX_VALUE_CGROUP_V2 {
139        return Some(MAX_VALUE);
140    }
141
142    let quota = quota.parse::<i64>().ok()?;
143
144    let period = fields[1].trim();
145    let period = period.parse::<i64>().ok()?;
146
147    // Return the cpu limit in millicores.
148    Some(quota * 1000 / period)
149}

151#[cfg(test)]
152mod tests {
153    use super::*;
154
155    #[test]
156    fn test_read_value_from_file() {
157        assert_eq!(
158            read_value_from_file(Path::new("testdata").join("memory.max")).unwrap(),
159            100000
160        );
161        assert_eq!(
162            read_value_from_file(Path::new("testdata").join("memory.max.unlimited")).unwrap(),
163            MAX_VALUE
164        );
165        assert_eq!(read_value_from_file(Path::new("non_existent_file")), None);
166    }
167
168    #[test]
169    fn test_get_cgroup_v2_cpu_limit() {
170        assert_eq!(
171            get_cgroup_v2_cpu_limit(Path::new("testdata").join("cpu.max")).unwrap(),
172            1500
173        );
174        assert_eq!(
175            get_cgroup_v2_cpu_limit(Path::new("testdata").join("cpu.max.unlimited")).unwrap(),
176            MAX_VALUE
177        );
178        assert_eq!(
179            get_cgroup_v2_cpu_limit(Path::new("non_existent_file")),
180            None
181        );
182    }
