52pub struct Instance {
53    frontend: Frontend,
54    // Keep the logging guard to prevent the worker from being dropped.
55    _guard: Vec<WorkerGuard>,
56}
57
58pub const APP_NAME: &str = "greptime-frontend";
59
60impl Instance {
61    pub fn new(frontend: Frontend, _guard: Vec<WorkerGuard>) -> Self {
62        Self { frontend, _guard }
63    }
64
65    pub fn inner(&self) -> &Frontend {
66        &self.frontend
67    }
68
69    pub fn mut_inner(&mut self) -> &mut Frontend {
70        &mut self.frontend
71    }
72}
73
74#[async_trait]
75impl App for Instance {
76    fn name(&self) -> &str {
77        APP_NAME
78    }
79
80    async fn start(&mut self) -> Result<()> {
81        let plugins = self.frontend.instance.plugins().clone();
82        plugins::start_frontend_plugins(plugins)
83            .await
84            .context(error::StartFrontendSnafu)?;
85
86        self.frontend
87            .start()
88            .await
89            .context(error::StartFrontendSnafu)
90    }
91
92    async fn stop(&mut self) -> Result<()> {
93        self.frontend
94            .shutdown()
95            .await
96            .context(error::ShutdownFrontendSnafu)
97    }
98}

100#[derive(Parser)]
101pub struct Command {
102    #[clap(subcommand)]
103    subcmd: SubCommand,
104}
105
106impl Command {
107    pub async fn build(&self, opts: FrontendOptions) -> Result<Instance> {
108        self.subcmd.build(opts).await
109    }
110
111    pub fn load_options(&self, global_options: &GlobalOptions) -> Result<FrontendOptions> {
112        self.subcmd.load_options(global_options)
113    }
114}
115
116#[derive(Parser)]
117enum SubCommand {
118    Start(StartCommand),
119}
120
121impl SubCommand {
122    async fn build(&self, opts: FrontendOptions) -> Result<Instance> {
123        match self {
124            SubCommand::Start(cmd) => cmd.build(opts).await,
125        }
126    }
127
128    fn load_options(&self, global_options: &GlobalOptions) -> Result<FrontendOptions> {
129        match self {
130            SubCommand::Start(cmd) => cmd.load_options(global_options),
131        }
132    }
133}

135#[derive(Debug, Default, Parser)]
136pub struct StartCommand {
137    /// The address to bind the gRPC server.
138    #[clap(long, alias = "rpc-addr")]
139    rpc_bind_addr: Option<String>,
140    /// The address advertised to the metasrv, and used for connections from outside the host.
141    /// If left empty or unset, the server will automatically use the IP address of the first network interface
142    /// on the host, with the same port number as the one specified in `rpc_bind_addr`.
143    #[clap(long, alias = "rpc-hostname")]
144    rpc_server_addr: Option<String>,
145    #[clap(long)]
146    http_addr: Option<String>,
147    #[clap(long)]
148    http_timeout: Option<u64>,
149    #[clap(long)]
150    mysql_addr: Option<String>,
151    #[clap(long)]
152    postgres_addr: Option<String>,
153    #[clap(short, long)]
154    config_file: Option<String>,
155    #[clap(short, long)]
156    influxdb_enable: Option<bool>,
157    #[clap(long, value_delimiter = ',', num_args = 1..)]
158    metasrv_addrs: Option<Vec<String>>,
159    #[clap(long)]
160    tls_mode: Option<TlsMode>,
161    #[clap(long)]
162    tls_cert_path: Option<String>,
163    #[clap(long)]
164    tls_key_path: Option<String>,
165    #[clap(long)]
166    user_provider: Option<String>,
167    #[clap(long)]
168    disable_dashboard: Option<bool>,
169    #[clap(long, default_value = "GREPTIMEDB_FRONTEND")]
170    env_prefix: String,
171}
172
173impl StartCommand {
174    fn load_options(&self, global_options: &GlobalOptions) -> Result<FrontendOptions> {
175        let mut opts = FrontendOptions::load_layered_options(
176            self.config_file.as_deref(),
177            self.env_prefix.as_ref(),
178        )
179        .context(error::LoadLayeredConfigSnafu)?;
180
181        self.merge_with_cli_options(global_options, &mut opts)?;
182
183        Ok(opts)
184    }
185
186    // The precedence order is: cli > config file > environment variables > default values.
187    fn merge_with_cli_options(
188        &self,
189        global_options: &GlobalOptions,
190        opts: &mut FrontendOptions,
191    ) -> Result<()> {
192        let opts = &mut opts.component;
193
194        if let Some(dir) = &global_options.log_dir {
195            opts.logging.dir.clone_from(dir);
196        }
197
198        // If the logging dir is not set, use the default logs dir in the data home.
199        if opts.logging.dir.is_empty() {
200            opts.logging.dir = Path::new(DEFAULT_DATA_HOME)
201                .join(DEFAULT_LOGGING_DIR)
202                .to_string_lossy()
203                .to_string();
204        }
205
206        if global_options.log_level.is_some() {
207            opts.logging.level.clone_from(&global_options.log_level);
208        }
209
210        opts.tracing = TracingOptions {
211            #[cfg(feature = "tokio-console")]
212            tokio_console_addr: global_options.tokio_console_addr.clone(),
213        };
214
215        let tls_opts = TlsOption::new(
216            self.tls_mode.clone(),
217            self.tls_cert_path.clone(),
218            self.tls_key_path.clone(),
219        );
220
221        if let Some(addr) = &self.http_addr {
222            opts.http.addr.clone_from(addr);
223        }
224
225        if let Some(http_timeout) = self.http_timeout {
226            opts.http.timeout = Duration::from_secs(http_timeout)
227        }
228
229        if let Some(disable_dashboard) = self.disable_dashboard {
230            opts.http.disable_dashboard = disable_dashboard;
231        }
232
233        if let Some(addr) = &self.rpc_bind_addr {
234            opts.grpc.bind_addr.clone_from(addr);
235            opts.grpc.tls = tls_opts.clone();
236        }
237
238        if let Some(addr) = &self.rpc_server_addr {
239            opts.grpc.server_addr.clone_from(addr);
240        }
241
242        if let Some(addr) = &self.mysql_addr {
243            opts.mysql.enable = true;
244            opts.mysql.addr.clone_from(addr);
245            opts.mysql.tls = tls_opts.clone();
246        }
247
248        if let Some(addr) = &self.postgres_addr {
249            opts.postgres.enable = true;
250            opts.postgres.addr.clone_from(addr);
251            opts.postgres.tls = tls_opts;
252        }
253
254        if let Some(enable) = self.influxdb_enable {
255            opts.influxdb.enable = enable;
256        }
257
258        if let Some(metasrv_addrs) = &self.metasrv_addrs {
259            opts.meta_client
260                .get_or_insert_with(MetaClientOptions::default)
261                .metasrv_addrs
262                .clone_from(metasrv_addrs);
263        }
264
265        if let Some(user_provider) = &self.user_provider {
266            opts.user_provider = Some(user_provider.clone());
267        }
268
269        Ok(())
270    }
271
272    async fn build(&self, opts: FrontendOptions) -> Result<Instance> {
273        common_runtime::init_global_runtimes(&opts.runtime);
274
275        let guard = common_telemetry::init_global_logging(
276            APP_NAME,
277            &opts.component.logging,
278            &opts.component.tracing,
279            opts.component.node_id.clone(),
280            opts.component.slow_query.as_ref(),
281        );
282
283        log_versions(version(), short_version(), APP_NAME);
284        create_resource_limit_metrics(APP_NAME);
285
286        info!("Frontend start command: {:#?}", self);
287        info!("Frontend options: {:#?}", opts);
288
289        let plugin_opts = opts.plugins;
290        let mut opts = opts.component;
291        opts.grpc.detect_server_addr();
292        let mut plugins = Plugins::new();
293        plugins::setup_frontend_plugins(&mut plugins, &plugin_opts, &opts)
294            .await
295            .context(error::StartFrontendSnafu)?;
296
297        set_default_timezone(opts.default_timezone.as_deref()).context(error::InitTimezoneSnafu)?;
298
299        let meta_client_options = opts
300            .meta_client
301            .as_ref()
302            .context(error::MissingConfigSnafu {
303                msg: "'meta_client'",
304            })?;
305
306        let cache_max_capacity = meta_client_options.metadata_cache_max_capacity;
307        let cache_ttl = meta_client_options.metadata_cache_ttl;
308        let cache_tti = meta_client_options.metadata_cache_tti;
309
310        let meta_client = meta_client::create_meta_client(
311            MetaClientType::Frontend,
312            meta_client_options,
313            Some(&plugins),
314        )
315        .await
316        .context(error::MetaClientInitSnafu)?;
317
318        // TODO(discord9): add helper function to ease the creation of cache registry&such
319        let cached_meta_backend =
320            CachedKvBackendBuilder::new(Arc::new(MetaKvBackend::new(meta_client.clone())))
321                .cache_max_capacity(cache_max_capacity)
322                .cache_ttl(cache_ttl)
323                .cache_tti(cache_tti)
324                .build();
325        let cached_meta_backend = Arc::new(cached_meta_backend);
326
327        // Builds cache registry
328        let layered_cache_builder = LayeredCacheRegistryBuilder::default().add_cache_registry(
329            CacheRegistryBuilder::default()
330                .add_cache(cached_meta_backend.clone())
331                .build(),
332        );
333        let fundamental_cache_registry =
334            build_fundamental_cache_registry(Arc::new(MetaKvBackend::new(meta_client.clone())));
335        let layered_cache_registry = Arc::new(
336            with_default_composite_cache_registry(
337                layered_cache_builder.add_cache_registry(fundamental_cache_registry),
338            )
339            .context(error::BuildCacheRegistrySnafu)?
340            .build(),
341        );
342
343        let information_extension =
344            Arc::new(DistributedInformationExtension::new(meta_client.clone()));
345        let catalog_manager = KvBackendCatalogManager::new(
346            information_extension,
347            cached_meta_backend.clone(),
348            layered_cache_registry.clone(),
349            None,
350        );
351
352        let executor = HandlerGroupExecutor::new(vec![
353            Arc::new(ParseMailboxMessageHandler),
354            Arc::new(InvalidateCacheHandler::new(layered_cache_registry.clone())),
355        ]);
356
357        let heartbeat_task = HeartbeatTask::new(
358            &opts,
359            meta_client.clone(),
360            opts.heartbeat.clone(),
361            Arc::new(executor),
362        );
363        let heartbeat_task = Some(heartbeat_task);
364
365        // frontend to datanode need not timeout.
366        // Some queries are expected to take long time.
367        let channel_config = ChannelConfig {
368            timeout: None,
369            tcp_nodelay: opts.datanode.client.tcp_nodelay,
370            connect_timeout: Some(opts.datanode.client.connect_timeout),
371            ..Default::default()
372        };
373        let client = NodeClients::new(channel_config);
374
375        let instance = FrontendBuilder::new(
376            opts.clone(),
377            cached_meta_backend.clone(),
378            layered_cache_registry.clone(),
379            catalog_manager,
380            Arc::new(client),
381            meta_client,
382        )
383        .with_plugin(plugins.clone())
384        .with_local_cache_invalidator(layered_cache_registry)
385        .try_build()
386        .await
387        .context(error::StartFrontendSnafu)?;
388        let instance = Arc::new(instance);
389
390        let export_metrics_task = ExportMetricsTask::try_new(&opts.export_metrics, Some(&plugins))
391            .context(error::ServersSnafu)?;
392
393        let servers = Services::new(opts, instance.clone(), plugins)
394            .build()
395            .context(error::StartFrontendSnafu)?;
396
397        let frontend = Frontend {
398            instance,
399            servers,
400            heartbeat_task,
401            export_metrics_task,
402        };
403
404        Ok(Instance::new(frontend, guard))
405    }

544    #[test]
545    fn test_config_precedence_order() {
546        let mut file = create_named_temp_file();
547        let toml_str = r#"
548            [http]
549            addr = "127.0.0.1:4000"
550
551            [meta_client]
552            timeout = "3s"
553            connect_timeout = "5s"
554            tcp_nodelay = true
555
556            [mysql]
557            addr = "127.0.0.1:4002"
558        "#;
559        write!(file, "{}", toml_str).unwrap();
560
561        let env_prefix = "FRONTEND_UT";
562        temp_env::with_vars(
563            [
564                (
565                    // mysql.addr = 127.0.0.1:14002
566                    [
567                        env_prefix.to_string(),
568                        "mysql".to_uppercase(),
569                        "addr".to_uppercase(),
570                    ]
571                    .join(ENV_VAR_SEP),
572                    Some("127.0.0.1:14002"),
573                ),
574                (
575                    // mysql.runtime_size = 11
576                    [
577                        env_prefix.to_string(),
578                        "mysql".to_uppercase(),
579                        "runtime_size".to_uppercase(),
580                    ]
581                    .join(ENV_VAR_SEP),
582                    Some("11"),
583                ),
584                (
585                    // http.addr = 127.0.0.1:24000
586                    [
587                        env_prefix.to_string(),
588                        "http".to_uppercase(),
589                        "addr".to_uppercase(),
590                    ]
591                    .join(ENV_VAR_SEP),
592                    Some("127.0.0.1:24000"),
593                ),
594                (
595                    // meta_client.metasrv_addrs = 127.0.0.1:3001,127.0.0.1:3002,127.0.0.1:3003
596                    [
597                        env_prefix.to_string(),
598                        "meta_client".to_uppercase(),
599                        "metasrv_addrs".to_uppercase(),
600                    ]
601                    .join(ENV_VAR_SEP),
602                    Some("127.0.0.1:3001,127.0.0.1:3002,127.0.0.1:3003"),
603                ),
604            ],
605            || {
606                let command = StartCommand {
607                    config_file: Some(file.path().to_str().unwrap().to_string()),
608                    http_addr: Some("127.0.0.1:14000".to_string()),
609                    env_prefix: env_prefix.to_string(),
610                    ..Default::default()
611                };
612
613                let fe_opts = command.load_options(&Default::default()).unwrap().component;
614
615                // Should be read from env, env > default values.
616                assert_eq!(fe_opts.mysql.runtime_size, 11);
617                assert_eq!(
618                    fe_opts.meta_client.unwrap().metasrv_addrs,
619                    vec![
620                        "127.0.0.1:3001".to_string(),
621                        "127.0.0.1:3002".to_string(),
622                        "127.0.0.1:3003".to_string()
623                    ]
624                );
625
626                // Should be read from config file, config file > env > default values.
627                assert_eq!(fe_opts.mysql.addr, "127.0.0.1:4002");
628
629                // Should be read from cli, cli > config file > env > default values.
630                assert_eq!(fe_opts.http.addr, "127.0.0.1:14000");
631
632                // Should be default value.
633                assert_eq!(fe_opts.grpc.bind_addr, GrpcOptions::default().bind_addr);
634            },
635        );
636    }
