48/// A log store backed by Kafka.
49#[derive(Debug)]
50pub struct KafkaLogStore {
51    /// The manager of topic clients.
52    client_manager: ClientManagerRef,
53    /// The max size of a batch.
54    max_batch_bytes: usize,
55    /// The consumer wait timeout.
56    consumer_wait_timeout: Duration,
57    /// Ignore missing entries during read WAL.
58    overwrite_entry_start_id: bool,
59    /// High watermark for all topics.
60    ///
61    /// Represents the offset of the last record in each topic. This is used to track
62    /// the latest available data in Kafka topics.
63    ///
64    /// The high watermark is updated in two ways:
65    /// - Automatically when the producer successfully commits data to Kafka
66    /// - Periodically by the [HighWatermarkManager](crate::kafka::high_watermark_manager::HighWatermarkManager).
67    ///
68    /// This shared map allows multiple components to access the latest high watermark
69    /// information without needing to query Kafka directly.
70    high_watermark: Arc<DashMap<Arc<KafkaProvider>, u64>>,
71}
72
73impl KafkaLogStore {
74    /// Tries to create a Kafka log store.
75    pub async fn try_new(
76        config: &DatanodeKafkaConfig,
77        global_index_collector: Option<GlobalIndexCollector>,
78    ) -> Result<Self> {
79        let high_watermark = Arc::new(DashMap::new());
80        let client_manager = Arc::new(
81            ClientManager::try_new(config, global_index_collector, high_watermark.clone()).await?,
82        );
83        let high_watermark_manager = HighWatermarkManager::new(
84            DEFAULT_HIGH_WATERMARK_UPDATE_INTERVAL,
85            high_watermark.clone(),
86            client_manager.clone(),
87        );
88        high_watermark_manager.run().await;
89
90        Ok(Self {
91            client_manager,
92            max_batch_bytes: config.max_batch_bytes.as_bytes() as usize,
93            consumer_wait_timeout: config.consumer_wait_timeout,
94            overwrite_entry_start_id: config.overwrite_entry_start_id,
95            high_watermark,
96        })
97    }
98}

100fn build_entry(
101    data: Vec<u8>,
102    entry_id: EntryId,
103    region_id: RegionId,
104    provider: &Provider,
105    max_data_size: usize,
106) -> Entry {
107    if data.len() <= max_data_size {
108        Entry::Naive(NaiveEntry {
109            provider: provider.clone(),
110            region_id,
111            entry_id,
112            data,
113        })
114    } else {
115        let parts = data
116            .chunks(max_data_size)
117            .map(|s| s.into())
118            .collect::<Vec<_>>();
119        let num_parts = parts.len();
120
121        let mut headers = Vec::with_capacity(num_parts);
122        headers.push(MultiplePartHeader::First);
123        headers.extend((1..num_parts - 1).map(MultiplePartHeader::Middle));
124        headers.push(MultiplePartHeader::Last);
125
126        Entry::MultiplePart(MultiplePartEntry {
127            provider: provider.clone(),
128            region_id,
129            entry_id,
130            headers,
131            parts,
132        })
133    }
134}

165    /// Appends a batch of entries and returns a response containing a map where the key is a region id
166    /// while the value is the id of the last successfully written entry of the region.
167    async fn append_batch(&self, entries: Vec<Entry>) -> Result<AppendBatchResponse> {
168        metrics::METRIC_KAFKA_APPEND_BATCH_BYTES_TOTAL.inc_by(
169            entries
170                .iter()
171                .map(|entry| entry.estimated_size())
172                .sum::<usize>() as u64,
173        );
174        let _timer = metrics::METRIC_KAFKA_APPEND_BATCH_ELAPSED.start_timer();
175
176        if entries.is_empty() {
177            return Ok(AppendBatchResponse::default());
178        }
179
180        let region_ids = entries
181            .iter()
182            .map(|entry| entry.region_id())
183            .collect::<HashSet<_>>();
184        let mut region_grouped_records: HashMap<RegionId, (OrderedBatchProducerRef, Vec<_>)> =
185            HashMap::with_capacity(region_ids.len());
186        let mut region_to_provider = HashMap::with_capacity(region_ids.len());
187        for entry in entries {
188            let provider = entry.provider().as_kafka_provider().with_context(|| {
189                error::InvalidProviderSnafu {
190                    expected: KafkaProvider::type_name(),
191                    actual: entry.provider().type_name(),
192                }
193            })?;
194            region_to_provider.insert(entry.region_id(), provider.clone());
195            let region_id = entry.region_id();
196            match region_grouped_records.entry(region_id) {
197                std::collections::hash_map::Entry::Occupied(mut slot) => {
198                    slot.get_mut().1.extend(convert_to_kafka_records(entry)?);
199                }
200                std::collections::hash_map::Entry::Vacant(slot) => {
201                    let producer = self
202                        .client_manager
203                        .get_or_insert(provider)
204                        .await?
205                        .producer()
206                        .clone();
207
208                    slot.insert((producer, convert_to_kafka_records(entry)?));
209                }
210            }
211        }
212
213        let mut region_grouped_result_receivers = Vec::with_capacity(region_ids.len());
214        for (region_id, (producer, records)) in region_grouped_records {
215            // Safety: `KafkaLogStore::entry` will ensure that the
216            // `Record`'s `approximate_size` must be less or equal to `max_batch_bytes`.
217            region_grouped_result_receivers
218                .push((region_id, producer.produce(region_id, records).await?))
219        }
220
221        let region_grouped_max_offset =
222            try_join_all(region_grouped_result_receivers.into_iter().map(
223                |(region_id, receiver)| async move {
224                    receiver.wait().await.map(|offset| (region_id, offset))
225                },
226            ))
227            .await?;
228
229        // Updates the high watermark offset of the last record in the topic.
230        for (region_id, offset) in &region_grouped_max_offset {
231            // Safety: `region_id` is always valid.
232            let provider = region_to_provider.get(region_id).unwrap();
233            self.high_watermark.insert(provider.clone(), *offset);
234        }
235
236        Ok(AppendBatchResponse {
237            last_entry_ids: region_grouped_max_offset.into_iter().collect(),
238        })
239    }

241    /// Creates a new `EntryStream` to asynchronously generates `Entry` with entry ids.
242    /// Returns entries belonging to `provider`, starting from `entry_id`.
243    async fn read(
244        &self,
245        provider: &Provider,
246        mut entry_id: EntryId,
247        index: Option<WalIndex>,
248    ) -> Result<SendableEntryStream<'static, Entry, Self::Error>> {
249        let provider = provider
250            .as_kafka_provider()
251            .with_context(|| InvalidProviderSnafu {
252                expected: KafkaProvider::type_name(),
253                actual: provider.type_name(),
254            })?;
255
256        let _timer = metrics::METRIC_KAFKA_READ_ELAPSED.start_timer();
257
258        // Gets the client associated with the topic.
259        let client = self
260            .client_manager
261            .get_or_insert(provider)
262            .await?
263            .client()
264            .clone();
265
266        if self.overwrite_entry_start_id {
267            let start_offset =
268                client
269                    .get_offset(OffsetAt::Earliest)
270                    .await
271                    .context(GetOffsetSnafu {
272                        topic: &provider.topic,
273                    })?;
274
275            if entry_id as i64 <= start_offset {
276                warn!(
277                "The entry_id: {} is less than start_offset: {}, topic: {}. Overwriting entry_id with start_offset",
278                entry_id, start_offset, &provider.topic
279            );
280
281                entry_id = start_offset as u64;
282            }
283        }
284
285        // Gets the offset of the latest record in the topic. Actually, it's the latest record of the single partition in the topic.
286        // The read operation terminates when this record is consumed.
287        // Warning: the `get_offset` returns the end offset of the latest record. For our usage, it should be decremented.
288        // See: https://kafka.apache.org/36/javadoc/org/apache/kafka/clients/consumer/KafkaConsumer.html#endOffsets(java.util.Collection)
289        let end_offset = client
290            .get_offset(OffsetAt::Latest)
291            .await
292            .context(GetOffsetSnafu {
293                topic: &provider.topic,
294            })?;
295
296        let region_indexes = if let (Some(index), Some(collector)) =
297            (index, self.client_manager.global_index_collector())
298        {
299            collector
300                .read_remote_region_index(index.location_id, provider, index.region_id, entry_id)
301                .await?
302        } else {
303            None
304        };
305
306        let Some(iterator) = build_region_wal_index_iterator(
307            entry_id,
308            end_offset as u64,
309            region_indexes,
310            self.max_batch_bytes,
311            MIN_BATCH_WINDOW_SIZE,
312        ) else {
313            let range = entry_id..end_offset as u64;
314            warn!("No new entries in range {:?} of ns {}", range, provider);
315            return Ok(futures_util::stream::empty().boxed());
316        };
317
318        debug!("Reading entries with {:?} of ns {}", iterator, provider);
319
320        // Safety: Must be ok.
321        let mut stream_consumer = ConsumerBuilder::default()
322            .client(client)
323            // Safety: checked before.
324            .buffer(RecordsBuffer::new(iterator))
325            .max_batch_size(self.max_batch_bytes)
326            .max_wait_ms(self.consumer_wait_timeout.as_millis() as u32)
327            .build()
328            .unwrap();
329
330        // A buffer is used to collect records to construct a complete entry.
331        let mut entry_records: HashMap<RegionId, Vec<Record>> = HashMap::new();
332        let provider = provider.clone();
333        let stream = async_stream::stream!({
334            while let Some(consume_result) = stream_consumer.next().await {
335                // Each next on the stream consumer produces a `RecordAndOffset` and a high watermark offset.
336                // The `RecordAndOffset` contains the record data and its start offset.
337                // The high watermark offset is the offset of the last record plus one.
338                let (record_and_offset, high_watermark) =
339                    consume_result.context(ConsumeRecordSnafu {
340                        topic: &provider.topic,
341                    })?;
342                let (kafka_record, offset) = (record_and_offset.record, record_and_offset.offset);
343
344                metrics::METRIC_KAFKA_READ_BYTES_TOTAL
345                    .inc_by(kafka_record.approximate_size() as u64);
346
347                debug!(
348                    "Read a record at offset {} for topic {}, high watermark: {}",
349                    offset, provider.topic, high_watermark
350                );
351
352                // Ignores no-op records.
353                if kafka_record.value.is_none() {
354                    if check_termination(offset, end_offset) {
355                        if let Some(entries) = remaining_entries(&provider, &mut entry_records) {
356                            yield Ok(entries);
357                        }
358                        break;
359                    }
360                    continue;
361                }
362
363                let record = Record::try_from(kafka_record)?;
364                // Tries to construct an entry from records consumed so far.
365                if let Some(mut entry) = maybe_emit_entry(&provider, record, &mut entry_records)? {
366                    // We don't rely on the EntryId generated by mito2.
367                    // Instead, we use the offset return from Kafka as EntryId.
368                    // Therefore, we MUST overwrite the EntryId with RecordOffset.
369                    entry.set_entry_id(offset as u64);
370                    yield Ok(vec![entry]);
371                }
372
373                if check_termination(offset, end_offset) {
374                    if let Some(entries) = remaining_entries(&provider, &mut entry_records) {
375                        yield Ok(entries);
376                    }
377                    break;
378                }
379            }
380        });
381        Ok(Box::pin(stream))
382    }

399    /// Marks all entries with ids `<=entry_id` of the given `namespace` as obsolete,
400    /// so that the log store can safely delete those entries. This method does not guarantee
401    /// that the obsolete entries are deleted immediately.
402    async fn obsolete(
403        &self,
404        provider: &Provider,
405        region_id: RegionId,
406        entry_id: EntryId,
407    ) -> Result<()> {
408        if let Some(collector) = self.client_manager.global_index_collector() {
409            let provider = provider
410                .as_kafka_provider()
411                .with_context(|| InvalidProviderSnafu {
412                    expected: KafkaProvider::type_name(),
413                    actual: provider.type_name(),
414                })?;
415            collector.truncate(provider, region_id, entry_id).await?;
416        }
417        Ok(())
418    }

420    /// Returns the highest entry id of the specified topic in remote WAL.
421    fn high_watermark(&self, provider: &Provider) -> Result<EntryId> {
422        let provider = provider
423            .as_kafka_provider()
424            .with_context(|| InvalidProviderSnafu {
425                expected: KafkaProvider::type_name(),
426                actual: provider.type_name(),
427            })?;
428
429        let high_watermark = self
430            .high_watermark
431            .get(provider)
432            .as_deref()
433            .copied()
434            .unwrap_or(0);
435
436        Ok(high_watermark)
437    }

445fn check_termination(offset: i64, end_offset: i64) -> bool {
446    // Terminates the stream if the entry with the end offset was read.
447    if offset >= end_offset {
448        debug!("Stream consumer terminates at offset {}", offset);
449        // There must have no records when the stream terminates.
450        true
451    } else {
452        false
453    }

541        provider: &Provider,
542        num_entries: usize,

667                    generate_entries(&logstore, &provider, 20, region_id, data_size_kb * 1024),
668                )
669            })
670            .collect::<HashMap<RegionId, Vec<_>>>();
671
672        let mut all_entries = region_entries
673            .values()
674            .flatten()
675            .cloned()
676            .collect::<Vec<_>>();
677        assert_matches!(all_entries[0], Entry::MultiplePart(_));
678        all_entries.shuffle(&mut rand::rng());
679
680        let response = logstore.append_batch(all_entries.clone()).await.unwrap();
681        // 5 region
682        assert_eq!(response.last_entry_ids.len(), 5);
683        let got_entries = logstore
684            .read(&provider, 0, None)
685            .await
686            .unwrap()
687            .try_collect::<Vec<_>>()
688            .await
689            .unwrap()
690            .into_iter()
691            .flatten()
692            .collect::<Vec<_>>();
693        for (region_id, _) in region_entries {
694            let expected_entries = all_entries
695                .iter()
696                .filter(|entry| entry.region_id() == region_id)
697                .cloned()
698                .collect::<Vec<_>>();
699            let mut actual_entries = got_entries
700                .iter()
701                .filter(|entry| entry.region_id() == region_id)
702                .cloned()
703                .collect::<Vec<_>>();
704            actual_entries
705                .iter_mut()
706                .for_each(|entry| entry.set_entry_id(0));
707            assert_eq!(expected_entries, actual_entries);
708        }
709        let high_wathermark = logstore.high_watermark(&provider).unwrap();
710        assert_eq!(high_wathermark, (data_size_kb as u64 / 8 + 1) * 20 * 5 - 1);
711    }
