114    ///
115    /// If it's true, the region failover will be allowed even if the local WAL is used.
116    /// Note that this option is not recommended to be set to true, because it may lead to data loss during failover.
117    pub allow_region_failover_on_local_wal: bool,
118    pub grpc: GrpcOptions,
119    /// The HTTP server options.
120    pub http: HttpOptions,
121    /// The logging options.
122    pub logging: LoggingOptions,

126    pub failure_detector: PhiAccrualFailureDetectorOptions,
127    /// The datanode options.
128    pub datanode: DatanodeClientOptions,
129    /// Whether to enable telemetry.
130    pub enable_telemetry: bool,
131    /// The data home directory.
132    pub data_home: String,
133    /// The WAL options.
134    pub wal: MetasrvWalConfig,
135    /// The metrics export options.
136    pub export_metrics: ExportMetricsOption,
137    /// The store key prefix. If it is not empty, all keys in the store will be prefixed with it.
138    /// This is useful when multiple metasrv clusters share the same store.
139    pub store_key_prefix: String,
140    /// The max operations per txn
141    ///
142    /// This value is usually limited by which store is used for the `KvBackend`.
143    /// For example, if using etcd, this value should ensure that it is less than
144    /// or equal to the `--max-txn-ops` option value of etcd.
145    ///
146    /// TODO(jeremy): Currently, this option only affects the etcd store, but it may
147    /// also affect other stores in the future. In other words, each store needs to

149    /// potentially block other operations.
150    pub max_txn_ops: usize,
151    /// The factor that determines how often statistics should be flushed,
152    /// based on the number of received heartbeats. When the number of heartbeats

155    /// The tracing options.
156    pub tracing: TracingOptions,
157    /// The datastore for kv metadata.
158    pub backend: BackendImpl,
159    #[cfg(any(feature = "pg_kvbackend", feature = "mysql_kvbackend"))]
160    /// Table name of rds kv backend.
161    pub meta_table_name: String,
162    #[cfg(feature = "pg_kvbackend")]
163    /// Lock id for meta kv election. Only effect when using pg_kvbackend.
164    pub meta_election_lock_id: u64,
165    #[serde(with = "humantime_serde")]
166    pub node_max_idle_time: Duration,
167}

363pub struct MetaStateHandler {
364    subscribe_manager: Option<SubscriptionManagerRef>,
365    greptimedb_telemetry_task: Arc<GreptimeDBTelemetryTask>,
366    leader_cached_kv_backend: Arc<LeaderCachedKvBackend>,
367    leadership_change_notifier: LeadershipChangeNotifier,
368    state: StateRef,
369}
370
371impl MetaStateHandler {
372    pub async fn on_leader_start(&self) {
373        self.state.write().unwrap().next_state(become_leader(false));
374
375        if let Err(e) = self.leader_cached_kv_backend.load().await {
376            error!(e; "Failed to load kv into leader cache kv store");
377        } else {
378            self.state.write().unwrap().next_state(become_leader(true));
379        }
380
381        self.leadership_change_notifier
382            .notify_on_leader_start()
383            .await;
384
385        self.greptimedb_telemetry_task.should_report(true);
386    }
387
388    pub async fn on_leader_stop(&self) {
389        self.state.write().unwrap().next_state(become_follower());
390
391        self.leadership_change_notifier
392            .notify_on_leader_stop()
393            .await;
394
395        // Suspends reporting.
396        self.greptimedb_telemetry_task.should_report(false);
397
398        if let Some(sub_manager) = self.subscribe_manager.clone() {
399            info!("Leader changed, un_subscribe all");
400            if let Err(e) = sub_manager.unsubscribe_all() {
401                error!(e; "Failed to un_subscribe all");
402            }
403        }
404    }
405}

442impl Metasrv {
443    pub async fn try_start(&self) -> Result<()> {
444        if self
445            .started
446            .compare_exchange(false, true, Ordering::AcqRel, Ordering::Acquire)
447            .is_err()
448        {
449            warn!("Metasrv already started");
450            return Ok(());
451        }
452
453        let handler_group_builder =
454            self.handler_group_builder
455                .lock()
456                .unwrap()
457                .take()
458                .context(error::UnexpectedSnafu {
459                    violated: "expected heartbeat handler group builder",
460                })?;
461        *self.handler_group.write().unwrap() = Some(Arc::new(handler_group_builder.build()?));
462
463        // Creates default schema if not exists
464        self.table_metadata_manager
465            .init()
466            .await
467            .context(InitMetadataSnafu)?;
468
469        if let Some(election) = self.election() {
470            let procedure_manager = self.procedure_manager.clone();
471            let in_memory = self.in_memory.clone();
472            let leader_cached_kv_backend = self.leader_cached_kv_backend.clone();
473            let subscribe_manager = self.subscription_manager();
474            let mut rx = election.subscribe_leader_change();
475            let greptimedb_telemetry_task = self.greptimedb_telemetry_task.clone();
476            greptimedb_telemetry_task
477                .start()
478                .context(StartTelemetryTaskSnafu)?;
479
480            // Builds leadership change notifier.
481            let mut leadership_change_notifier = LeadershipChangeNotifier::default();
482            leadership_change_notifier.add_listener(self.wal_options_allocator.clone());
483            leadership_change_notifier
484                .add_listener(Arc::new(ProcedureManagerListenerAdapter(procedure_manager)));
485            leadership_change_notifier.add_listener(Arc::new(NodeExpiryListener::new(
486                self.options.node_max_idle_time,
487                self.in_memory.clone(),
488            )));
489            if let Some(region_supervisor_ticker) = &self.region_supervisor_ticker {
490                leadership_change_notifier.add_listener(region_supervisor_ticker.clone() as _);
491            }
492            if let Some(wal_prune_ticker) = &self.wal_prune_ticker {
493                leadership_change_notifier.add_listener(wal_prune_ticker.clone() as _);
494            }
495            if let Some(customizer) = self.plugins.get::<LeadershipChangeNotifierCustomizerRef>() {
496                customizer.customize(&mut leadership_change_notifier);
497            }
498
499            let state_handler = MetaStateHandler {
500                greptimedb_telemetry_task,
501                subscribe_manager,
502                state: self.state.clone(),
503                leader_cached_kv_backend: leader_cached_kv_backend.clone(),
504                leadership_change_notifier,
505            };
506            let _handle = common_runtime::spawn_global(async move {
507                loop {
508                    match rx.recv().await {
509                        Ok(msg) => {
510                            in_memory.reset();
511                            leader_cached_kv_backend.reset();
512                            info!("Leader's cache has bean cleared on leader change: {msg}");
513                            match msg {
514                                LeaderChangeMessage::Elected(_) => {
515                                    state_handler.on_leader_start().await;
516                                }
517                                LeaderChangeMessage::StepDown(leader) => {
518                                    error!("Leader :{:?} step down", leader);
519
520                                    state_handler.on_leader_stop().await;
521                                }
522                            }
523                        }
524                        Err(RecvError::Closed) => {
525                            error!("Not expected, is leader election loop still running?");
526                            break;
527                        }
528                        Err(RecvError::Lagged(_)) => {
529                            break;
530                        }
531                    }
532                }
533
534                state_handler.on_leader_stop().await;
535            });
536
537            // Register candidate and keep lease in background.
538            {
539                let election = election.clone();
540                let started = self.started.clone();
541                let node_info = self.node_info();
542                let _handle = common_runtime::spawn_global(async move {
543                    while started.load(Ordering::Acquire) {
544                        let res = election.register_candidate(&node_info).await;
545                        if let Err(e) = res {
546                            warn!(e; "Metasrv register candidate error");
547                        }
548                    }
549                });
550            }
551
552            // Campaign
553            {
554                let election = election.clone();
555                let started = self.started.clone();
556                let _handle = common_runtime::spawn_global(async move {
557                    while started.load(Ordering::Acquire) {
558                        let res = election.campaign().await;
559                        if let Err(e) = res {
560                            warn!(e; "Metasrv election error");
561                        }
562                        election.reset_campaign().await;
563                        info!("Metasrv re-initiate election");
564                    }
565                    info!("Metasrv stopped");
566                });
567            }
568        } else {
569            warn!(
570                "Ensure only one instance of Metasrv is running, as there is no election service."
571            );
572
573            if let Err(e) = self.wal_options_allocator.start().await {
574                error!(e; "Failed to start wal options allocator");
575            }
576            // Always load kv into cached kv store.
577            self.leader_cached_kv_backend
578                .load()
579                .await
580                .context(KvBackendSnafu)?;
581            self.procedure_manager
582                .start()
583                .await
584                .context(StartProcedureManagerSnafu)?;
585        }
586
587        info!("Metasrv started");
588
589        Ok(())
590    }

592    pub async fn shutdown(&self) -> Result<()> {
593        if self
594            .started
595            .compare_exchange(true, false, Ordering::AcqRel, Ordering::Acquire)
596            .is_err()
597        {
598            warn!("Metasrv already stopped");
599            return Ok(());
600        }
601
602        self.procedure_manager
603            .stop()
604            .await
605            .context(StopProcedureManagerSnafu)?;
606
607        info!("Metasrv stopped");
608
609        Ok(())
610    }
