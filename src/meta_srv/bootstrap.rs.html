98impl MetasrvInstance {
99    pub async fn new(
100        opts: MetasrvOptions,
101        plugins: Plugins,
102        metasrv: Metasrv,
103    ) -> Result<MetasrvInstance> {
104        let http_server = HttpServerBuilder::new(opts.http.clone())
105            .with_metrics_handler(MetricsHandler)
106            .with_greptime_config_options(opts.to_toml().context(error::TomlFormatSnafu)?)
107            .build();
108
109        let metasrv = Arc::new(metasrv);
110        // put metasrv into plugins for later use
111        plugins.insert::<Arc<Metasrv>>(metasrv.clone());
112        let export_metrics_task = ExportMetricsTask::try_new(&opts.export_metrics, Some(&plugins))
113            .context(error::InitExportMetricsTaskSnafu)?;
114        Ok(MetasrvInstance {
115            metasrv,
116            http_server,
117            opts,
118            signal_sender: None,
119            plugins,
120            export_metrics_task,
121            serve_state: Default::default(),
122            bind_addr: None,
123        })
124    }
125
126    pub async fn start(&mut self) -> Result<()> {
127        self.metasrv.try_start().await?;
128
129        if let Some(t) = self.export_metrics_task.as_ref() {
130            t.start(None).context(error::InitExportMetricsTaskSnafu)?
131        }
132
133        let (tx, rx) = mpsc::channel::<()>(1);
134
135        self.signal_sender = Some(tx);
136
137        let mut router = router(self.metasrv.clone());
138        if let Some(configurator) = self.metasrv.plugins().get::<ConfiguratorRef>() {
139            router = configurator.config_grpc(router);
140        }
141
142        let (serve_state_tx, serve_state_rx) = oneshot::channel();
143
144        let socket_addr =
145            bootstrap_metasrv_with_router(&self.opts.grpc.bind_addr, router, serve_state_tx, rx)
146                .await?;
147        self.bind_addr = Some(socket_addr);
148
149        let addr = self.opts.http.addr.parse().context(error::ParseAddrSnafu {
150            addr: &self.opts.http.addr,
151        })?;
152        self.http_server
153            .start(addr)
154            .await
155            .context(error::StartHttpSnafu)?;
156
157        *self.serve_state.lock().await = Some(serve_state_rx);
158        Ok(())
159    }
160
161    pub async fn shutdown(&self) -> Result<()> {
162        if let Some(mut rx) = self.serve_state.lock().await.take() {
163            if let Ok(Err(err)) = rx.try_recv() {
164                common_telemetry::error!(err; "Metasrv start failed")
165            }
166        }
167        if let Some(signal) = &self.signal_sender {
168            signal
169                .send(())
170                .await
171                .context(error::SendShutdownSignalSnafu)?;
172        }
173        self.metasrv.shutdown().await?;
174        self.http_server
175            .shutdown()
176            .await
177            .context(error::ShutdownServerSnafu {
178                server: self.http_server.name(),
179            })?;
180        Ok(())
181    }
182
183    pub fn plugins(&self) -> Plugins {
184        self.plugins.clone()
185    }
186
187    pub fn get_inner(&self) -> &Metasrv {
188        &self.metasrv
189    }
190    pub fn bind_addr(&self) -> &Option<SocketAddr> {
191        &self.bind_addr
192    }
193}

195pub async fn bootstrap_metasrv_with_router(
196    bind_addr: &str,
197    router: Router,
198    serve_state_tx: oneshot::Sender<Result<()>>,
199    mut shutdown_rx: Receiver<()>,
200) -> Result<SocketAddr> {
201    let listener = TcpListener::bind(bind_addr)
202        .await
203        .context(error::TcpBindSnafu { addr: bind_addr })?;
204
205    let real_bind_addr = listener
206        .local_addr()
207        .context(error::TcpBindSnafu { addr: bind_addr })?;
208
209    info!("gRPC server is bound to: {}", real_bind_addr);
210
211    let incoming =
212        TcpIncoming::from_listener(listener, true, None).context(error::TcpIncomingSnafu)?;
213
214    let _handle = common_runtime::spawn_global(async move {
215        let result = router
216            .serve_with_incoming_shutdown(incoming, async {
217                let _ = shutdown_rx.recv().await;
218            })
219            .await
220            .inspect_err(|err| common_telemetry::error!(err;"Failed to start metasrv"))
221            .context(error::StartGrpcSnafu);
222        let _ = serve_state_tx.send(result);
223    });
224
225    Ok(real_bind_addr)
226}

228#[macro_export]
229macro_rules! add_compressed_service {
230    ($builder:expr, $server:expr) => {
231        $builder.add_service(
232            $server
233                .accept_compressed(CompressionEncoding::Gzip)
234                .accept_compressed(CompressionEncoding::Zstd)
235                .send_compressed(CompressionEncoding::Gzip)
236                .send_compressed(CompressionEncoding::Zstd),
237        )
238    };
239}

241pub fn router(metasrv: Arc<Metasrv>) -> Router {
242    let mut router = tonic::transport::Server::builder().accept_http1(true); // for admin services
243    let router = add_compressed_service!(router, HeartbeatServer::from_arc(metasrv.clone()));
244    let router = add_compressed_service!(router, StoreServer::from_arc(metasrv.clone()));
245    let router = add_compressed_service!(router, ClusterServer::from_arc(metasrv.clone()));
246    let router = add_compressed_service!(router, ProcedureServiceServer::from_arc(metasrv.clone()));
247    router.add_service(admin::make_admin_service(metasrv))
248}

250pub async fn metasrv_builder(
251    opts: &MetasrvOptions,
252    plugins: Plugins,
253    kv_backend: Option<KvBackendRef>,
254) -> Result<MetasrvBuilder> {
255    let (mut kv_backend, election) = match (kv_backend, &opts.backend) {
256        (Some(kv_backend), _) => (kv_backend, None),
257        (None, BackendImpl::MemoryStore) => (Arc::new(MemoryKvBackend::new()) as _, None),
258        (None, BackendImpl::EtcdStore) => {
259            let etcd_client = create_etcd_client(&opts.store_addrs).await?;
260            let kv_backend = EtcdStore::with_etcd_client(etcd_client.clone(), opts.max_txn_ops);
261            let election = EtcdElection::with_etcd_client(
262                &opts.grpc.server_addr,
263                etcd_client,
264                opts.store_key_prefix.clone(),
265            )
266            .await?;
267
268            (kv_backend, Some(election))
269        }
270        #[cfg(feature = "pg_kvbackend")]
271        (None, BackendImpl::PostgresStore) => {
272            use std::time::Duration;
273
274            use common_meta::distributed_time_constants::POSTGRES_KEEP_ALIVE_SECS;
275
276            use crate::election::rds::postgres::ElectionPgClient;
277
278            let candidate_lease_ttl = Duration::from_secs(CANDIDATE_LEASE_SECS);
279            let execution_timeout = Duration::from_secs(META_LEASE_SECS);
280            let statement_timeout = Duration::from_secs(META_LEASE_SECS);
281            let meta_lease_ttl = Duration::from_secs(META_LEASE_SECS);
282
283            let mut cfg = Config::new();
284            cfg.keepalives = Some(true);
285            cfg.keepalives_idle = Some(Duration::from_secs(POSTGRES_KEEP_ALIVE_SECS));
286            // We use a separate pool for election since we need a different session keep-alive idle time.
287            let pool = create_postgres_pool_with(&opts.store_addrs, cfg).await?;
288
289            let election_client =
290                ElectionPgClient::new(pool, execution_timeout, meta_lease_ttl, statement_timeout)?;
291            let election = PgElection::with_pg_client(
292                opts.grpc.server_addr.clone(),
293                election_client,
294                opts.store_key_prefix.clone(),
295                candidate_lease_ttl,
296                meta_lease_ttl,
297                &opts.meta_table_name,
298                opts.meta_election_lock_id,
299            )
300            .await?;
301
302            let pool = create_postgres_pool(&opts.store_addrs).await?;
303            let kv_backend = PgStore::with_pg_pool(pool, &opts.meta_table_name, opts.max_txn_ops)
304                .await
305                .context(error::KvBackendSnafu)?;
306
307            (kv_backend, Some(election))
308        }
309        #[cfg(feature = "mysql_kvbackend")]
310        (None, BackendImpl::MysqlStore) => {
311            let pool = create_mysql_pool(&opts.store_addrs).await?;
312            let kv_backend =
313                MySqlStore::with_mysql_pool(pool, &opts.meta_table_name, opts.max_txn_ops)
314                    .await
315                    .context(error::KvBackendSnafu)?;
316            // Since election will acquire a lock of the table, we need a separate table for election.
317            let election_table_name = opts.meta_table_name.clone() + "_election";
318            let election_client = create_mysql_client(opts).await?;
319            let election = MySqlElection::with_mysql_client(
320                opts.grpc.server_addr.clone(),
321                election_client,
322                opts.store_key_prefix.clone(),
323                CANDIDATE_LEASE_SECS,
324                META_LEASE_SECS,
325                &election_table_name,
326            )
327            .await?;
328            (kv_backend, Some(election))
329        }
330    };
331
332    if !opts.store_key_prefix.is_empty() {
333        info!(
334            "using chroot kv backend with prefix: {prefix}",
335            prefix = opts.store_key_prefix
336        );
337        kv_backend = Arc::new(ChrootKvBackend::new(
338            opts.store_key_prefix.clone().into_bytes(),
339            kv_backend,
340        ))
341    }
342
343    let in_memory = Arc::new(MemoryKvBackend::new()) as ResettableKvBackendRef;
344
345    let node_excluder = plugins
346        .get::<NodeExcluderRef>()
347        .unwrap_or_else(|| Arc::new(Vec::new()) as NodeExcluderRef);
348    let selector = if let Some(selector) = plugins.get::<SelectorRef>() {
349        info!("Using selector from plugins");
350        selector
351    } else {
352        let selector = match opts.selector {
353            SelectorType::LoadBased => Arc::new(LoadBasedSelector::new(
354                RegionNumsBasedWeightCompute,
355                node_excluder,
356            )) as SelectorRef,
357            SelectorType::LeaseBased => {
358                Arc::new(LeaseBasedSelector::new(node_excluder)) as SelectorRef
359            }
360            SelectorType::RoundRobin => Arc::new(RoundRobinSelector::new(
361                SelectTarget::Datanode,
362                node_excluder,
363            )) as SelectorRef,
364        };
365        info!(
366            "Using selector from options, selector type: {}",
367            opts.selector.as_ref()
368        );
369        selector
370    };
371
372    Ok(MetasrvBuilder::new()
373        .options(opts.clone())
374        .kv_backend(kv_backend)
375        .in_memory(in_memory)
376        .selector(selector)
377        .election(election)
378        .plugins(plugins))
379}

381pub async fn create_etcd_client(store_addrs: &[String]) -> Result<Client> {
382    let etcd_endpoints = store_addrs
383        .iter()
384        .map(|x| x.trim())
385        .filter(|x| !x.is_empty())
386        .collect::<Vec<_>>();
387    Client::connect(&etcd_endpoints, None)
388        .await
389        .context(error::ConnectEtcdSnafu)
390}

400#[cfg(feature = "pg_kvbackend")]
401/// Creates a pool for the Postgres backend.
402///
403/// It only use first store addr to create a pool, and use the given config to create a pool.
404pub async fn create_postgres_pool_with(
405    store_addrs: &[String],
406    mut cfg: Config,
407) -> Result<deadpool_postgres::Pool> {
408    let postgres_url = store_addrs.first().context(error::InvalidArgumentsSnafu {
409        err_msg: "empty store addrs",
410    })?;
411    cfg.url = Some(postgres_url.to_string());
412    let pool = cfg
413        .create_pool(Some(Runtime::Tokio1), NoTls)
414        .context(error::CreatePostgresPoolSnafu)?;
415    Ok(pool)
416}

418#[cfg(feature = "mysql_kvbackend")]
419async fn setup_mysql_options(store_addrs: &[String]) -> Result<MySqlConnectOptions> {
420    let mysql_url = store_addrs.first().context(error::InvalidArgumentsSnafu {
421        err_msg: "empty store addrs",
422    })?;
423    // Avoid `SET` commands in sqlx
424    let opts: MySqlConnectOptions = mysql_url
425        .parse()
426        .context(error::ParseMySqlUrlSnafu { mysql_url })?;
427    let opts = opts
428        .no_engine_substitution(false)
429        .pipes_as_concat(false)
430        .timezone(None)
431        .set_names(false);
432    Ok(opts)
433}
